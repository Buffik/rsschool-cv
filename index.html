function createDreamTeam(members) {
    let tmpArr = []
    let result = ''
    if (!Array.isArray(members)) {
        return false
    }
    members.forEach((e) => {
        if (typeof(e) === 'string') {
        tmpArr.push(e.replace(/\s+/g, ''))
        } 
    })
    
    if (tmpArr.length === 0) {
        return false
    }
    
    for (let i = 0; i < tmpArr.length; i++) {
        result += tmpArr[i][0]
    }
    
    return result.toUpperCase().split('').sort().join('')
    }
